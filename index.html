// --- CONFIGURATION --- const GD_CONFIG = { SPEED: 350, GRAVITY: 2500, JUMP_FORCE: -700, SHIP_THRUST: -1200 }; class GameScene extends Phaser.Scene { constructor() { super('GameScene'); } create() { // 1. Player Setup this.player = this.physics.add.sprite(100, 450, 'player'); this.player.setGravityY(GD_CONFIG.GRAVITY); this.mode = 'CUBE'; // Initial mode // 2. Groups for Objects this.platforms = this.physics.add.staticGroup(); this.orbs = this.physics.add.group(); this.pads = this.physics.add.group(); // 3. Simple Level Layout (Placeholders) this.platforms.create(400, 580, 'ground').setScale(20, 1).refreshBody(); let yellowOrb = this.orbs.create(500, 450, 'orb_yellow'); let jumpPad = this.pads.create(800, 560, 'pad_yellow'); // 4. Input Listener this.input.on('pointerdown', () => this.handleAction()); this.spaceBar = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.SPACE); // 5. Collisions this.physics.add.collider(this.player, this.platforms); // Pad Logic: Instant Trigger this.physics.add.overlap(this.player, this.pads, (p, pad) => { p.setVelocityY(GD_CONFIG.JUMP_FORCE * 1.5); }); } handleAction() { // Check for Orbs first let touchingOrb = false; this.physics.overlap(this.player, this.orbs, (p, orb) => { p.setVelocityY(GD_CONFIG.JUMP_FORCE); touchingOrb = true; }); // If not touching an orb, do standard jump if (!touchingOrb && this.player.body.touching.down) { this.player.setVelocityY(GD_CONFIG.JUMP_FORCE); } } update() { // Constant Horizontal Movement this.player.setVelocityX(GD_CONFIG.SPEED); // Mode: Ship Logic (Hold to fly) if (this.mode === 'SHIP' && (this.input.activePointer.isDown || this.spaceBar.isDown)) { this.player.setAccelerationY(GD_CONFIG.SHIP_THRUST); } else { this.player.setAccelerationY(0); } } }
